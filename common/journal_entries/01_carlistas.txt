je_carlist_victory = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_historical_content

	complete = {
        custom_tooltip = {
			text = win_civil_war
			hidden_trigger = {
				NOT = { exists = c:SPA}
			}
		}	
	}

	on_weekly_pulse = {
		effect = {
			trigger_event = { id = esp.14 }
		}
	}

	on_complete = {
		if = {
			limit = { has_modifier = guerra_total }
			remove_modifier = guerra_total
		}
		change_tag = SPA
		set_capital = STATE_TOLEDO

		if = {
			limit = { 
				any_scope_building = {
					has_modifier = division_auxiliar_fra
				}
			}
			random_scope_building = {
				limit = {
					has_modifier = division_auxiliar_fra
				}
				remove_modifier = division_auxiliar_fra
			}
			hidden_effect = {
				c:FRA = { trigger_event = { id = caw.6 days = 1 } }
			}
		}

		if = {
			limit = { 
				any_scope_building = {
					has_modifier = legion_auxiliar_gbr
				}
			}
			random_scope_building = {
				limit = {
					has_modifier = legion_auxiliar_gbr
				}
				remove_modifier = legion_auxiliar_gbr
			}
			hidden_effect = {
				c:GBR = { trigger_event = { id = caw.6 days = 1 } }
			}
		}

		custom_tooltip = {
			text = je_carlist_victory_1
			add_journal_entry = { type = je_deuda_esp }
			add_journal_entry = { type = je_infraestructura_esp }
			add_journal_entry = { type = je_falta_capital_esp }
		}

		hidden_effect = { 
			trigger_event = { id = car.2 }
			set_variable = vicotria_carlista
			if = {
				limit = { EXISTS = c:PHI }
				create_diplomatic_pact = {
					country = c:PHI
					type = puppet
				}
			}
			every_scope_general = {
				limit = { is_busy = yes }
				set_character_busy_and_immortal = no
			}
		}
	}

    should_be_pinned_by_default = yes
	weight = 100
}
