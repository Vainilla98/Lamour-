namespace = caw

caw.1 = {
	type = country_event
	placement = ROOT

	title = caw.1.t
	desc = caw.1.d
	flavor = caw.1.f

	event_image = {
		video = "middleeast_battlefield_trenches"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	duration = 1

	immediate = {
		c:SPA = { save_scope_as = spa_scope }
		c:CAR = { save_scope_as = car_scope }
	}

	option = {
		name = caw.1.a
		default_option = yes
		change_relations = {
			country = c:SPA
			value = 60
		}
		add_modifier = {
			name = apoyo_isabelino
			months = normal_modifier_time
		}
		custom_tooltip = caw.1_tt
		if = {
			limit = { ROOT = c:FRA }
			scope:spa_scope = {
				trigger_event = { id = caw.2 popup = no days = 5 }
			}
		}
		else = {
			scope:spa_scope = {
				trigger_event = { id = caw.3 popup = no days = 7 } 
			}
		}
		ai_chance = {
			base = 1
			modifier = {
				trigger = {
					OR = {
						has_attitude = {
							who = scope:spa_scope
							attitude = protective
						}
						has_attitude = {
							who = scope:spa_scope
							attitude = genial
						}
						has_attitude = {
							who = scope:spa_scope
							attitude = cooperative
						}
						has_attitude = {
							who = scope:spa_scope
							attitude = conciliatory
						}
					}
				}
				add = 1
			}
		}
	}
	option = {
		name = caw.1.b
		if = {
			limit = { ROOT = c:FRA }
			scope:spa_scope = {
				trigger_event = { id = caw.4 popup = no days = 5 } 
			}
		}
		else = {
			scope:spa_scope = {
				trigger_event = { id = caw.5 popup = no days = 7 } 
			}
		}
		ai_chance = {
			base = 1
			modifier = {
				trigger = {
					OR = {
						has_attitude = {
							who = scope:spa_scope
							attitude = belligerent
						}
						has_attitude = {
							who = scope:spa_scope
							attitude = antagonistic
						}
					}
				}
				add = 1
			}
		}
	}	
}

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	duration = 1

	immediate = {
		c:FRA = { save_scope_as = fra_scope }
	}

	option = {
		name = caw.2.a
		custom_label = caw.2.tt
		create_military_formation = {
			type = army
			hq_region = sr:region_iberia
			name = "División Auxiliar francesa"

			combat_unit = {
				type = unit_type:combat_unit_type_line_infantry
				state_region = s:STATE_VALENCIA 
				count = 1
			}
		}
		s:STATE_VALENCIA.region_state:SPA = {
			create_pop = {
				culture = spanish
				pop_type = soldiers 
				religion = catholic
				size = 1000
			}
			
		}
	}	
}
